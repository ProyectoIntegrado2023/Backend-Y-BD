--Creacion de Tablas con sus Atributos
--Son 26 tablas hoy creacion 26/10/23

CREATE TABLE PROYECTO (
    id_PROYECTO number,
    NOMBRE VARCHAR2(200),
    CODIGO VARCHAR2(200),
    COORDINADOR VARCHAR2(200),
    FECHA_INICIO DATE,
    FECHA_FIN DATE,
    OBJETIVOS VARCHAR2(1000),
    DESCRIPCION VARCHAR2(2000),
    PRESUPUESTO NUMBER,
    INFORME_FINAL VARCHAR2(200),
    id_TIPO_CONVENIO number,
    id_SEMESTRE number,
    id_ESTADO number,
    id_CICLO number, 
    id_ESCUELA number,
    id_CURSO_ARTICULADO number,
    CONSTRAINT nombre_pk PRIMARY KEY (id_PROYECTO)
);

CREATE TABLE PLANTILLA (
    id_PLANTILLA number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    URL VARCHAR2(200),
    id_TIPO_CONVENIO number,
     
    CONSTRAINT PLANTILLAPK PRIMARY KEY (id_PLANTILLA)
);

CREATE TABLE FACULTAD (
    id_FACULTAD number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT FACULTADPK PRIMARY KEY (id_FACULTAD)
);

CREATE TABLE ESCUELA (
    id_ESCUELA number,
    id_FACULTAD number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT ESCUELAPK PRIMARY KEY (id_ESCUELA)
);

CREATE TABLE TIPO_DE_CONVENIO (
    id_TIPO_DE_CONVENIO number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT TIPO_DE_CONVENIOPK PRIMARY KEY (id_TIPO_DE_CONVENIO)
);

CREATE TABLE CURSO_ARTICULADO (
    id_CURSO_ARTICULADO number,
    id_DOCENTE number,
    id_ESCUELA number,
    id_PERSONA number,
    NOMBRE VARCHAR2(200), 
    ESTADO VARCHAR2(200),
    CONSTRAINT CURSO_ARTICULADOPK PRIMARY KEY (id_CURSO_ARTICULADO)
);

CREATE TABLE SEMESTRE (
    id_SEMESTRE number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT SEMESTREPK PRIMARY KEY (id_SEMESTRE)
);



CREATE TABLE ROL_SISTEMA (
    id_ROL_SISTEMA number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
     id_ESCUELA number,
    CONSTRAINT ROL_SISTEMAPK PRIMARY KEY (id_ROL_SISTEMA)
);


--FALTA VALORES
CREATE TABLE ACCESOS (
    id_ACCESOS number,
    ID_ACCESOS_PADRE NUMBER,
    NOMBRE VARCHAR2(200),
    URL  VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT ACCESOSPK PRIMARY KEY (id_ACCESOS)
);


CREATE TABLE ESTADO (
    id_ESTADO number,
    NOMBRE VARCHAR2(200),   
    CONSTRAINT ESTADOPK PRIMARY KEY (id_ESTADO)
);

CREATE TABLE ROL_NEGOCIO (
    id_ROL_NEGOCIO number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT ROL_NEGOCIOPK PRIMARY KEY (id_ROL_NEGOCIO)
);

CREATE TABLE CICLO (
    id_CICLO number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    id_ESCUELA number,
    CONSTRAINT CICLOPK PRIMARY KEY (id_CICLO)
);




CREATE TABLE GRUPO_UNIV (
    id_GRUPO_UNIV number,
    NOMBRE VARCHAR2(200),
    ESTADO VARCHAR2(200),
    CONSTRAINT GRUPO_UNIVPK PRIMARY KEY (id_GRUPO_UNIV)
);

CREATE TABLE DOCENTE (
    id_DOCENTE number,
    id_PERSONA number,
    ESTADO VARCHAR2(200),
    CONSTRAINT DOCENTEPK PRIMARY KEY (id_DOCENTE)
);


CREATE TABLE USUARIO_ACCESOS (
    id_USUARIO_ACCESOS number,
    id_USUARIO number,
    id_ACCESOS number,
    CONSTRAINT USUARIO_ACCESOSPK PRIMARY KEY (id_USUARIO_ACCESOS)
);

CREATE TABLE USUARIO (
    id_USUARIO number,
    USERNAME VARCHAR2(100),
    PASSWORD VARCHAR2(100),
    id_PERSONA number,
    ESTADO VARCHAR2(200),
    CONSTRAINT USUARIOPK PRIMARY KEY (id_USUARIO)
);

CREATE TABLE RECURSO (
    id_RECURSO number,
    NOMBRE VARCHAR2(100),
    URL VARCHAR2(100),
    ESTADO VARCHAR2(200),
    id_PROYECTO number,
    CONSTRAINT RECURSOPK PRIMARY KEY (id_RECURSO)
);


CREATE TABLE ROL_PROYECTO (
    id_ROL_PROYECTO number,
    HORA NUMBER,
    DESCRIPCION VARCHAR2(1000),
    id_ROL_NEGOCIO number,
    id_PROYECTO number,
   ESTADO VARCHAR2(200),
    
    CONSTRAINT ROL_PROYECTOPK PRIMARY KEY (id_ROL_PROYECTO)
);



CREATE TABLE ROL_PROYECTO_ENCARGADO (
    id_ROL_PROYECTO_ENCARGADO number,
    ESTADO VARCHAR2(200),
    id_ROL_PROYECTO number,
    id_ENCARGADO number,

    
    
    CONSTRAINT ROL_PROYECTO_ENCARGADOPK PRIMARY KEY (id_ROL_PROYECTO_ENCARGADO)
);




CREATE TABLE ACTIVIDAD_PARTICIPANTE (
    id_ACTIVIDAD_PARTICIPANTE number,
    ESTADO VARCHAR2(200),
    id_ACTIVIDAD NUMBER,
    id_PARTICIPANTE number,
   
    
    
    CONSTRAINT ACTIVIDAD_PARTICIPANTEPK PRIMARY KEY (id_ACTIVIDAD_PARTICIPANTE)
);


CREATE TABLE ROL_SISTEMA_ACCESOS (
    id_ROL_SISTEMA_ACCESOS number,
    id_ROL_SISTEMA NUMBER,
    id_ACCESOS number,
    ESTADO VARCHAR2(200),
    
    CONSTRAINT ROL_SISTEMA_ACCESOSPK PRIMARY KEY (id_ROL_SISTEMA_ACCESOS)
);


CREATE TABLE ESTUDIANTE (
    id_ESTUDIANTE number,
    id_PERSONA number,
    CODIGO VARCHAR2(30),
    HORAS_TOTALES  NUMBER,
    ESTADO VARCHAR2(200),
    id_CICLO number,
    id_GRUPO_UNIV number,
    
    
    CONSTRAINT ESTUDIANTEPK PRIMARY KEY (id_ESTUDIANTE)
);




CREATE TABLE ENCARGADO (
    id_ENCARGADO number,
    NOMBRE VARCHAR2(100),
    ROL VARCHAR2(100),
    ESTADO VARCHAR2(200),
     id_PARTICIPANTE number,  
    CONSTRAINT ENCARGADOPK PRIMARY KEY (id_ENCARGADO)
);
